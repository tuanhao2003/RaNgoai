##### 1 : Khang - Hệ thống chia nhóm trong lớp học #####
Tôi là một sinh viên năm thứ ba ngành công nghệ thông tin. Tôi đang làm một số bài tập giải thuật sử dụng ngôn ngữ JavaScript, cụ thể tôi đang phát triển một hàm nhằm giải quyết bài toán chia nhóm trong lớp học.
Hệ thống nhận vào một danh sách sinh viên, trong đó mỗi sinh viên có thông tin gồm MSSV, họ tên và lớp, sau đó phân chia thành các nhóm.
Mặc dù đoạn mã đã hoạt động, nhưng vẫn còn một số vấn đề như sau: thuật toán chạy chưa tối ưu, có thể mất nhiều thời gian khi số lượng sinh viên lớn, và việc quản lý bộ nhớ chưa hiệu quả. Tôi muốn cải thiện hiệu suất của thuật toán và tối ưu hóa cách thức phân nhóm.
Hãy giúp tôi tái cấu trúc lại đoạn mã bên dưới để giải quyết các vấn đề nêu trên. Các thay đổi và chỉnh sửa không được chú thích trong mã mà cần phải được tóm tắt ở cuối.
Đây là đoạn mã mà tôi đã viết:
```java
class StudentGroupAllocator {
    divideIntonhoms(sinhviens, sonhom) {
        if (sinhviens.length === 0 || sonhom <= 0) return [];

        sinhviens.sort(() => Math.random() - 0.5); // Xáo trộn danh sách để chia ngẫu nhiên

        let nhoms = Array.from({ length: sonhom }, () => []);
        let index = 0;

        for (const sinhvien of sinhviens) {
            nhoms[index].push(sinhvien);
            index = (index + 1) % sonhom; // Chia đều sinh viên vào các nhóm
        }

        return nhoms;
    }
}

const sinhviens = [
    { mssv: "123", name: "Nguyễn Văn A", class: "CNTT1" },
    { mssv: "456", name: "Trần Thị B", class: "CNTT2" },
    { mssv: "789", name: "Lê Văn C", class: "CNTT1" },
    { mssv: "101", name: "Phạm Thị D", class: "CNTT3" },
    { mssv: "112", name: "Hoàng Văn E", class: "CNTT2" },
    { mssv: "131", name: "Bùi Thị F", class: "CNTT3" }
];

const sonhom = 2;
const nhomSystem = new StudentGroupAllocator();
const nhoms = nhomSystem.divideIntonhoms(sinhviens, sonhom);

// In kết quả nhóm
nhoms.forEach((nhom, i) => {
    console.log(`Nhóm ${i + 1}:`, nhom);
});
